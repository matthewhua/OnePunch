1. 链表结构定义

// 链表节点结构
typedef struct LinkNode {
    int data;
    struct LinkNode *next;
} LinkNode;

typedef LinkNode* LinkList;

2. 初始化链表

int initList(LinkList *head) {
    *head = (LinkNode*)malloc(sizeof(LinkNode));
    if (*head == NULL) {
        return FALSE;
    }
    (*head)->next = NULL;
    return TRUE;
}

带头结点的单链表初始化：创建一个头结点，其next指针指向NULL。

时间复杂度： O(1)，空间复杂度： O(1)

3. 插入操作

int insertmy(LinkList *head, int pos, int value) {
    LinkNode *p, *newNode;
    int i;
    if (*head == NULL) return FALSE;
    p = *head;
    // 移动到插入位置的前一个节点
    for (i = 0; i < pos && p != NULL; i++) {
        p = p->next;
    }
    if (p == NULL) return FALSE;
    newNode = (LinkNode*)malloc(sizeof(LinkNode));
    newNode->data = value;
    newNode->next = p->next;
    p->next = newNode;
    return TRUE;
}
时间复杂度： O(n)，最坏情况下需要遍历到链表末尾

4. 删除操作

int removemy(LinkList *head, int pos, int *value) {
    LinkNode *p, *q;
    int i;
    if (*head == NULL || (*head)->next == NULL) return FALSE;
    p = *head;
    // 移动到删除位置的前一个节点
    for (i = 0; i < pos && p->next != NULL; i++) {
        p = p->next;
    }
    if (p->next == NULL) return FALSE;
    q = p->next;
    *value = q->data;
    p->next = q->next;
    free(q);
    return TRUE;
}

6. 链表逆置

int reverse(LinkList *head) {
    LinkNode *left, *middle, *right;
    left = (*head)->next;
    if (left != NULL) middle = left->next;
    left->next = NULL;
    while (middle != NULL) {
        right = middle->next;
        middle->next = left;
        left = middle;
        middle = right;
    }
    (*head)->next = left;
    return TRUE;
}
时间复杂度： O(n)，空间复杂度： O(1)
